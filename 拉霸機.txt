    <script>
        // --- 音效系統 ---
        const AudioContext = window.AudioContext || window.webkitAudioContext;
        let audioCtx = null;
        let isSoundEnabled = true;

        function initAudio() {
            if (!audioCtx) audioCtx = new AudioContext();
        }

        function playSound(freq, type, duration, vol = 0.1) {
            if (!isSoundEnabled || !audioCtx) return;
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            osc.type = type;
            osc.frequency.setValueAtTime(freq, audioCtx.currentTime);
            gain.gain.setValueAtTime(vol, audioCtx.currentTime);
            gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + duration);
            osc.connect(gain);
            gain.connect(audioCtx.destination);
            osc.start();
            osc.stop(audioCtx.currentTime + duration);
        }

        const soundEffects = {
            click: () => playSound(600, 'sine', 0.1, 0.1),
            tick: () => playSound(800, 'square', 0.05, 0.05),
            win: () => {
                setTimeout(() => playSound(523.25, 'sine', 0.5, 0.1), 0);   // C5
                setTimeout(() => playSound(659.25, 'sine', 0.5, 0.1), 150); // E5
                setTimeout(() => playSound(783.99, 'sine', 0.8, 0.1), 300); // G5
                setTimeout(() => playSound(1046.50, 'sine', 1.2, 0.1), 450); // C6
            }
        };

        // --- 抽籤邏輯 ---
        const originalWords = [
            "1. 縣政府", "2. 市政府", "3. 警察局", "4. 衛生所", 
            "5. 環保局", "6. 消防隊", "7. 區公所", "8. 國民運動中心",
            "9. 食予飽", "10. 穿予燒", "11. 寫予媠", "12. 坐予正",
            "13. 提予好", "14. 啉予焦", "15. 穿予整齊", "16. 看予清楚"
        ];

        let availableWords = [...originalWords];
        let history = [];
        let isSpinning = false;

        const slotStrip = document.getElementById('slotStrip');
        const spinBtn = document.getElementById('spinBtn');
        const resetBtn = document.getElementById('resetBtn');
        const historyList = document.getElementById('historyList');
        const remainingCount = document.getElementById('remainingCount');
        const winnerOverlay = document.getElementById('winnerOverlay');
        const winnerContent = document.getElementById('winnerContent');
        const winnerName = document.getElementById('winnerName');
        const closeOverlay = document.getElementById('closeOverlay');
        const soundToggle = document.getElementById('soundToggle');
        const soundOnIcon = document.getElementById('soundOnIcon');
        const soundOffIcon = document.getElementById('soundOffIcon');

        function initSlot() {
            slotStrip.innerHTML = '';
            for(let i=0; i<5; i++) {
                const item = document.createElement('div');
                item.className = 'slot-item';
                item.textContent = originalWords[Math.floor(Math.random() * originalWords.length)];
                slotStrip.appendChild(item);
            }
            updateUI();
        }

        function updateUI() {
            remainingCount.textContent = availableWords.length;
            spinBtn.disabled = availableWords.length === 0 || isSpinning;
            spinBtn.textContent = availableWords.length === 0 ? "抽獎完畢！" : "PULL / 拉霸！";

            if (history.length === 0) {
                historyList.innerHTML = '<div class="text-gray-500 text-center italic mt-10">尚無抽獎紀錄</div>';
            } else {
                historyList.innerHTML = history.map((word, index) => `
                    <div class="bg-white/10 p-3 rounded-lg flex items-center">
                        <span class="w-8 h-8 flex items-center justify-center bg-yellow-500 text-black font-bold rounded-full mr-3 text-sm">${history.length - index}</span>
                        <span class="font-bold">${word}</span>
                    </div>
                `).join('');
            }
        }

        function spin() {
            if (isSpinning || availableWords.length === 0) return;
            initAudio(); // 確保使用者互動後啟用音訊
            soundEffects.click();

            isSpinning = true;
            spinBtn.disabled = true;

            const winnerIndex = Math.floor(Math.random() * availableWords.length);
            const winner = availableWords[winnerIndex];

            const spinCount = 30; 
            const spinDuration = 3000; 
            
            let displayList = [];
            for (let i = 0; i < spinCount; i++) {
                displayList.push(originalWords[Math.floor(Math.random() * originalWords.length)]);
            }
            displayList[displayList.length - 1] = winner;

            slotStrip.style.transition = 'none';
            slotStrip.style.transform = 'translateY(0)';
            slotStrip.innerHTML = '';
            displayList.forEach(word => {
                const item = document.createElement('div');
                item.className = 'slot-item';
                item.textContent = word;
                slotStrip.appendChild(item);
            });

            slotStrip.offsetHeight;

            // 動畫與音效同步
            slotStrip.style.transition = `transform ${spinDuration}ms cubic-bezier(0.1, 0, 0.1, 1)`;
            const targetY = -( (displayList.length - 1) * 120 );
            slotStrip.style.transform = `translateY(${targetY}px)`;

            // 模擬旋轉時的滴答聲
            let startTime = null;
            function tick(timestamp) {
                if (!startTime) startTime = timestamp;
                const progress = timestamp - startTime;
                if (progress < spinDuration) {
                    // 隨進度變慢，減少滴答頻率
                    // 簡單處理：隨機或固定頻率，這裡用簡單間隔
                    if (Math.floor(progress / 100) % 1 === 0 && Math.random() > 0.5) {
                        soundEffects.tick();
                    }
                    requestAnimationFrame(tick);
                }
            }
            requestAnimationFrame(tick);

            setTimeout(() => {
                availableWords.splice(winnerIndex, 1);
                history.unshift(winner);
                showWinner(winner);
                isSpinning = false;
                updateUI();
            }, spinDuration + 200);
        }

        function showWinner(name) {
            soundEffects.win();
            winnerName.textContent = name;
            winnerOverlay.classList.remove('hidden');
            setTimeout(() => {
                winnerContent.classList.remove('scale-0');
                winnerContent.classList.add('scale-100');
            }, 10);
        }

        closeOverlay.onclick = () => {
            initAudio();
            soundEffects.click();
            winnerContent.classList.remove('scale-100');
            winnerContent.classList.add('scale-0');
            setTimeout(() => {
                winnerOverlay.classList.add('hidden');
            }, 300);
        };

        resetBtn.onclick = () => {
            initAudio();
            soundEffects.click();
            if (confirm("確定要重設所有語詞並重新開始嗎？")) {
                availableWords = [...originalWords];
                history = [];
                initSlot();
                updateUI();
            }
        };

        soundToggle.onclick = () => {
            initAudio();
            isSoundEnabled = !isSoundEnabled;
            soundOnIcon.classList.toggle('hidden', !isSoundEnabled);
            soundOffIcon.classList.toggle('hidden', isSoundEnabled);
            if(isSoundEnabled) soundEffects.click();
        };

        spinBtn.onclick = spin;
        window.onload = initSlot;
    </script>
